fast_pow.cpp — O(log n)
typedef long long ll; // a^n mod m dalam O(log n) ll power(ll a, ll n, ll m) { ll res = 1; a %= m; while (n > 0) { if (n & 1) res = res * a % m; a = a * a % m; n >>= 1; } return res; } // Contoh: power(2, 30, 1e9+7) = 1073741824

🔮 Sieve of Eratosthenes — Cari Semua Prima ≤ N

Algoritma paling efisien untuk mencari semua bilangan prima ≤ N. Kompleksitas O(N log log N) — jauh lebih cepat dari cek satu per satu.

Jumlah prima ≤ N ≈ N / ln(N)
Prime Number Theorem
Cari prima ≤ N
sieve.cpp — O(N log log N)
const int MAXN = 1e6+5; bool composite[MAXN]; vector<int> primes; void sieve(int n) { for (int i=2; i<=n; i++) { if (!composite[i]) { primes.push_back(i); for (ll j=(ll)i*i; j<=n; j+=i) composite[j] = true; } } } // Setelah dipanggil sieve(n): // primes = {2, 3, 5, 7, 11, 13, ...} // composite[x] = true jika x bukan prima

const int MAXN=1e6+5, MOD=1e9+7; ll fact[MAXN], inv_fact[MAXN]; void precompute() { fact[0] = 1; for(int i=1;i<MAXN;i++) fact[i] = fact[i-1]*i%MOD; inv_fact[MAXN-1] = power(fact[MAXN-1],MOD-2,MOD); for(int i=MAXN-2;i>=0;i--) inv_fact[i] = inv_fact[i+1]*(i+1)%MOD; } ll C(int n, int k) { if(k<0||k>n) return 0; return fact[n]*inv_fact[k]%MOD*inv_fact[n-k]%MOD; }
